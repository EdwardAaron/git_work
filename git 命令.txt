
git add FILE|DIR
	添加文件
git commmit 
	提交缓存区文件
git commit -a 
	提交修改过的和缓存过的文件
git rm FILE
	从版本库和本地中删除文件
git rm -f File|DIR
	强制删除
git rm --cached FILE|DIR
	只是从缓存区删除，本地文件保留
git clone url
	克隆仓库
git status
	检查文件状态
git status -s
	检查文件状态，输出简明信息
git diff
	比较修改与缓存文件之间的差异
git diff --cahed
	比较缓存和上次提交的差异
git mv FEIL1 FIEL2
	重命名文件
